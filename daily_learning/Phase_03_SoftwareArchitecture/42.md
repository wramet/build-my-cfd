### ✅ Implementing Gauss-Seidel Solver

Let's implement Gauss-Seidel, which converges much faster than Jacobi by using updated values immediately:

**File: `GaussSeidelSolver.H`**
```cpp
#ifndef GAUSS_SEIDEL_SOLVER_H
#define GAUSS_SEIDEL_SOLVER_H

#include "LinearSolver.H"

template<typename T>
class GaussSeidelSolver : public LinearSolver<T> {
private:
    scalar tolerance_;
    int maxIterations_;

public:
    GaussSeidelSolver(const SparseMatrix<T>& matrix, Vector<T>& solution, const Vector<T>& rhs)
        : LinearSolver<T>(matrix, solution, rhs) {}

    bool solve(scalar tolerance, int maxIterations) override {
        tolerance_ = tolerance;
        maxIterations_ = maxIterations;

        Vector<T> residual(this->solution_.size());
        scalar residualNorm = 0;
        int iteration = 0;

        do {
            // Gauss-Seidel update: use new values immediately
            for (int i = 0; i < this->matrix_.size(); ++i) {
                T sum = this->rhs_[i];

                // Lower triangular part (already updated)
                for (int j = 0; j < i; ++j) {
                    sum -= this->matrix_.coeff(i, j) * this->solution_[j];
                }

                // Upper triangular part (not yet updated)
                for (int j = i + 1; j < this->matrix_.size(); ++j) {
                    sum -= this->matrix_.coeff(i, j) * this->solution_[j];
                }

                // Update solution
                this->solution_[i] = sum / this->matrix_.coeff(i, i);
            }

            // Calculate residual
            this->matrix_.multiply residual = this->solution_;
            residual.axpy(-1.0, this->rhs_);
            residualNorm = residual.norm();

            iteration++;

        } while (residualNorm > tolerance && iteration < maxIterations);

        return (residualNorm <= tolerance);
    }

    std::string name() const override {
        return "GaussSeidel";
    }
};

#endif // GAUSS_SEIDEL_SOLVER_H
```

### ✅ Implementing Preconditioned Conjugate Gradient (PCG)

PCG is the most advanced iterative solver, ideal for symmetric positive definite systems:

**File: `PCGSolver.H`**
```cpp
#ifndef PCG_SOLVER_H
#define PCG_SOLVER_H

#include "LinearSolver.H"
#include <cmath>

template<typename T>
class PCGSolver : public LinearSolver<T> {
private:
    Vector<T> preconditioner_;  // Diagonal preconditioner
    Vector<T> residual_, searchDirection_, Ap_;

    scalar tolerance_;
    int maxIterations_;

    // Apply diagonal preconditioner: M^{-1} * x
    void applyPreconditioner(const Vector<T>& x, Vector<T>& result) const {
        for (int i = 0; i < x.size(); ++i) {
            result[i] = x[i] / preconditioner_[i];
        }
    }

    // Dot product for complex numbers (real part only)
    scalar dot(const Vector<T>& a, const Vector<T>& b) const {
        scalar result = 0;
        for (int i = 0; i < a.size(); ++i) {
            result += a[i] * b[i];  // Only real part for complex vectors
        }
        return result;
    }

public:
    PCGSolver(const SparseMatrix<T>& matrix, Vector<T>& solution, const Vector<T>& rhs)
        : LinearSolver<T>(matrix, solution, rhs),
          preconditioner_(matrix.size()),
          residual_(matrix.size()),
          searchDirection_(matrix.size()),
          Ap_(matrix.size())
    {
        // Setup diagonal preconditioner
        for (int i = 0; i < matrix.size(); ++i) {
            preconditioner_[i] = matrix.coeff(i, i);
            if (preconditioner_[i] < 1e-10) {  // Avoid division by zero
                preconditioner_[i] = 1.0;
            }
        }
    }

    bool solve(scalar tolerance, int maxIterations) override {
        tolerance_ = tolerance;
        maxIterations_ = maxIterations;

        // Initial residual
        this->matrix_.multiply residual = this->solution_;
        residual.axpy(-1.0, this->rhs_);

        // Apply preconditioner
        applyPreconditioner(residual, searchDirection_);

        scalar initialError = std::sqrt(dot(searchDirection_, residual));
        if (initialError < tolerance) {
            return true;  // Already converged
        }

        scalar alpha = 0, beta = 0;
        int iteration = 0;

        do {
            // Calculate Ap = A * p
            this->matrix_.multiply Ap = searchDirection_;

            // Calculate step size: alpha = (r^T * r) / (p^T * Ap)
            numerator = dot(residual, searchDirection_);
            denominator = dot(searchDirection_, Ap);
            alpha = numerator / denominator;

            // Update solution: x = x + alpha * p
            this->solution_.axpy(alpha, searchDirection_);

            // Update residual: r = r - alpha * Ap
            residual.axpy(-alpha, Ap);

            // Check convergence
            scalar error = std::sqrt(dot(residual, residual));
            if (error < tolerance) {
                return true;
            }

            // Update search direction: beta = (r_new^T * r_new) / (r_old^T * r_old)
            beta = dot(residual, residual) / numerator;

            // Update search direction: p = M^{-1} * r + beta * p
            Vector<T> temp(residual.size());
            applyPreconditioner(residual, temp);
            searchDirection_.beta = temp;
            searchDirection_.axpy(beta, searchDirection_);

            iteration++;

        } while (iteration < maxIterations);

        return false;  // Failed to converge
    }

    std::string name() const override {
        return "PCG";
    }
};

#endif // PCG_SOLVER_H
```

```cpp
#ifndef SOLVER_FACTORY_H
#define SOLVER_FACTORY_H

#include "LinearSolver.H"
#include <memory>
#include <string>
#include <unordered_map>

template<typename T>
class SolverFactory {
private:
    using CreatorFunction = std::unique_ptr<LinearSolver<T>>(*)(const SparseMatrix<T>&, Vector<T>&, const Vector<T>&);

    static std::unordered_map<std::string, CreatorFunction>& registry() {
        static std::unordered_map<std::string, CreatorFunction> reg;
        return reg;
    }

public:
    static void registerSolver(const std::string& name, CreatorFunction creator) {
        registry()[name] = creator;
    }

    static std::unique_ptr<LinearSolver<T>> create(const std::string& name,
                                                  const SparseMatrix<T>& matrix,
                                                  Vector<T>& solution,
                                                  const Vector<T>& rhs) {
        auto it = registry().find(name);
        if (it != registry().end()) {
            return it->second(matrix, solution, rhs);
        }
        throw std::runtime_error("Unknown solver: " + name);
    }

    static std::vector<std::string> availableSolvers() {
        std::vector<std::string> solvers;
        for (const auto& pair : registry()) {
            solvers.push_back(pair.first);
        }
        return solvers;
    }
};

// Helper macro for solver registration
#define REGISTER_SOLVER(SolverName, TypeName) \
    namespace { \
        struct SolverName##Register { \
            SolverName##Register() { \
                SolverFactory<TypeName>::registerSolver(#SolverName, \
                    [](const SparseMatrix<TypeName>& matrix, Vector<TypeName>& solution, const Vector<TypeName>& rhs) -> std::unique_ptr<LinearSolver<TypeName>> { \
                        return std::make_unique<SolverName<TypeName>>(matrix, solution, rhs); \
                    }); \
            } \
        }; \
        static SolverName##Register registerSolver_##SolverName; \
    }
```

### 🎯 The Ultimate Test: Adding SOR Without Touching Existing Code

This is where our factory shines! Let's add Successive Over-Relaxation (SOR) solver - a powerful variant of Gauss-Seidel - without modifying any existing files.

**Step 1: Create SOR Solver (NEW FILE ONLY)**
```cpp
// SORSolver.H
#ifndef SOR_SOLVER_H
#define SOR_SOLVER_H

#include "LinearSolver.H"

template<typename T>
class SORSolver : public LinearSolver<T> {
private:
    scalar omega_;  // Relaxation parameter (0 < omega < 2)
    scalar tolerance_;
    int maxIterations_;

public:
    SORSolver(const SparseMatrix<T>& matrix, Vector<T>& solution, const Vector<T>& rhs, scalar omega = 1.5)
        : LinearSolver<T>(matrix, solution, rhs), omega_(omega) {}

    bool solve(scalar tolerance, int maxIterations) override {
        tolerance_ = tolerance;
        maxIterations_ = maxIterations;

        Vector<T> residual(this->solution_.size());
        scalar residualNorm = 0;
        int iteration = 0;

        do {
            // SOR iteration: x_i^{(k+1)} = (1-ω)x_i^{(k)} + ω/aii * [bi - Σaij*x_j^{(k+1)} - Σaij*x_j^{(k)}]
            for (int i = 0; i < this->matrix_.size(); ++i) {
                T sum = this->rhs_[i];
                T oldXi = this->solution_[i];

                // Lower triangular (already updated)
                for (int j = 0; j < i; ++j) {
                    sum -= this->matrix_.coeff(i, j) * this->solution_[j];
                }

                // Upper triangular (not yet updated)
                for (int j = i + 1; j < this->matrix_.size(); ++j) {
                    sum -= this->matrix_.coeff(i, j) * this->solution_[j];
                }

                // SOR update
                T newXi = sum / this->matrix_.coeff(i, i);
                this->solution_[i] = (1.0 - omega_) * oldXi + omega_ * newXi;
            }

            // Calculate residual
            this->matrix_.multiply residual = this->solution_;
            residual.axpy(-1.0, this->rhs_);
            residualNorm = residual.norm();

            iteration++;

        } while (residualNorm > tolerance && iteration < maxIterations);

        return (residualNorm <= tolerance);
    }

    std::string name() const override {
        return "SOR(" + std::to_string(omega_) + ")";
    }

    scalar getOmega() const { return omega_; }
    void setOmega(scalar omega) { omega_ = omega; }
};

#endif // SOR_SOLVER_H
```

## Part 3: Complete 1D Heat Equation Solution (โซล์ชันสมการความร้อน 1D ที่สมบูรณ์)

### 🎯 Problem Setup

Let's solve the steady-state 1D heat equation with Dirichlet boundary conditions:

$$
\frac{d}{dx}\left(k\frac{dT}{dx}\right) = 0, \quad T(0) = 0, \quad T(L) = 100
$$

With uniform conductivity $k = 1$ and $N = 100$ cells, this becomes:

$$
-\frac{T_{i-1} - 2T_i + T_{i+1}}{\Delta x^2} = 0
$$

### ✅ Matrix Assembly

**File: `HeatEquationAssembler.H`**
```cpp
#ifndef HEAT_EQUATION_ASSEMBLER_H
#define HEAT_EQUATION_ASSEMBLER_H

#include "SparseMatrix.H"
#include "Vector.H"

template<typename T>
class HeatEquationAssembler {
private:
    int N_;          // Number of cells
    scalar L_;       // Domain length
    scalar k_;       // Thermal conductivity
    scalar dx_;      // Cell size

public:
    HeatEquationAssembler(int N, scalar L = 1.0, scalar k = 1.0)
        : N_(N), L_(L), k_(k), dx_(L / (N + 1)) {}

    void assemble(SparseMatrix<T>& matrix, Vector<T>& rhs) const {
        matrix.resize(N_);
        rhs.resize(N_);

        // Interior points: -k/dx^2 * [T_{i-1} - 2T_i + T_{i+1}] = 0
        scalar coeff = -k_ / (dx_ * dx_);

        for (int i = 0; i < N_; ++i) {
            if (i == 0) {
                // First cell: T_0 = boundary condition
                matrix.insert(i, i, 1.0);
                rhs[i] = 0.0;
            } else if (i == N_ - 1) {
                // Last cell: T_{N-1} = boundary condition
                matrix.insert(i, i, 1.0);
                rhs[i] = 100.0;
            } else {
                // Interior cell
                matrix.insert(i, i-1, coeff);
                matrix.insert(i, i, -2.0 * coeff);
                matrix.insert(i, i+1, coeff);
                rhs[i] = 0.0;
            }
        }

        matrix.compress();
    }

    scalar getCellCenter(int i) const {
        return (i + 0.5) * dx_;
    }
};

#endif // HEAT_EQUATION_ASSEMBLER_H
```

**File: `heatEquationSolver.C`**
```cpp
#include "SolverFactory.H"
#include "HeatEquationAssembler.H"
#include "SparseMatrix.H"
#include "Vector.H"
#include <iostream>
#include <fstream>
#include <iomanip>

template<typename T>
struct SolverResults {
    std::string name;
    bool converged;
    int iterations;
    scalar finalResidual;
    scalar solveTime;
};

template<typename T>
SolverResults<T> runSolver(const std::string& solverName,
                           const SparseMatrix<T>& matrix,
                           Vector<T>& solution,
                           const Vector<T>& rhs,
                           scalar tolerance,
                           int maxIterations) {

    auto start = std::chrono::high_resolution_clock::now();

    auto solver = SolverFactory<T>::create(solverName, matrix, solution, rhs);

    bool converged = solver->solve(tolerance, maxIterations);

    auto end = std::chrono::high_resolution_clock::now();
    std::chrono::duration<scalar> duration = end - start;

    // Calculate final residual
    Vector<T> residual(solution.size());
    matrix.multiply residual = solution;
    residual.axpy(-1.0, rhs);
    scalar finalResidual = residual.norm();

    return {
        solverName,
        converged,
        solver->getIterationCount(),  // Assume we add this method
        finalResidual,
        duration.count()
    };
}

int main(int argc, char* argv[]) {
    // Problem parameters
    int N = 100;        // Number of cells
    scalar L = 1.0;     // Domain length
    scalar T_left = 0.0; // Left boundary temperature
    scalar T_right = 100.0; // Right boundary temperature

    std::cout << "=== 1D Heat Equation Solver ===" << std::endl;
    std::cout << "Grid cells: " << N << std::endl;
    std::cout << "Domain length: " << L << std::endl;
    std::cout << "BCs: T(0) = " << T_left << ", T(L) = " << T_right << std::endl;
    std::cout << std::endl;

    // Assemble system
    HeatEquationAssembler<double> assembler(N, L, 1.0);
    SparseMatrix<double> A;
    Vector<double> b;

    assembler.assemble(A, b);

    // Initialize solution (linear profile)
    Vector<double> T(N);
    for (int i = 0; i < N; ++i) {
        T[i] = T_left + (T_right - T_left) * (i + 0.5) / N;
    }

    // Dictionary parsing (simplified)
    auto solverConfigs = parseSolverDict("solversDict");

    // Run all configured solvers
    std::vector<SolverResults<double>> results;

    for (const auto& config : solverConfigs) {
        std::cout << "Running " << config.name << " solver..." << std::endl;

        auto result = runSolver(config.solver, A, T, b, config.tolerance, config.maxIterations);
        result.name = config.name;  // Use config name instead of solver name

        results.push_back(result);

        std::cout << "  Converged: " << (result.converged ? "Yes" : "No") << std::endl;
        std::cout << "  Iterations: " << result.iterations << std::endl;
        std::cout << "  Final residual: " << result.finalResidual << std::endl;
        std::cout << "  Solve time: " << result.solveTime << " s" << std::endl;
        std::cout << std::endl;
    }

    // Generate output file
    generateOutputFile(T, assembler, "temperature_profile.csv");

    return 0;
}
```

### ✅ Performance Analysis

#### Convergence Rates

The theoretical convergence rates for our solvers:

1. **Jacobi**: $\rho(G) = \cos(\pi h) \approx 1 - \frac{\pi^2 h^2}{2}$
   - Slowest convergence, suitable for small problems
   - Parallelizable (all updates independent)

2. **Gauss-Seidel**: $\rho(GS) = \rho(G)^2 \approx (1 - \frac{\pi^2 h^2}{2})^2$
   - Twice as fast as Jacobi
   - Sequential updates, not parallelizable

3. **SOR(ω)**: Optimal ω for model problems:
   - $\omega_{opt} = \frac{2}{1 + \sqrt{1 - \rho(G)^2}}$
   - For our case: $\omega_{opt} \approx 1.85$
   - Can be 2-3x faster than Gauss-Seidel

4. **PCG**: Optimal for SPD systems
   - Superlinear convergence
   - Best for high accuracy requirements

#### Computational Cost Analysis

```cpp
// Operation counts per iteration:
// Jacobi:    2N mult + N add + N div
// GS:        3N mult + 2N add + N div
// SOR(ω):    3N mult + 2N add + 2N div
// PCG:       4N mult + 3N add + N div + 2 dot products
```

When accounting for operation counts and convergence rates:

- **Low accuracy (1e-4)**: PCG > SOR > GS > Jacobi
- **Medium accuracy (1e-6)**: PCG > SOR > GS > Jacobi
- **High accuracy (1e-8)**: PCG > SOR > GS > Jacobi
- **Extreme accuracy (1e-10)**: PCG >> SOR > GS >> Jacobi

#### Memory Usage

All solvers have similar memory requirements:
- **Matrix storage**: O(N) for tridiagonal
- **Vectors**: O(N) for solution, residual, and working vectors
- **No additional storage** needed for any solver

### ✅ When to Use Which Solver

- SOR: O(N) iterations with smaller constant
- PCG: O(√N) iterations for ideal problems

## Part 5: Exercises & Reflection (แบบฝึกหัดและการสะท้อนคิด)

### 🎯 Conceptual Questions

#### 1. Why does Gauss-Seidel converge faster than Jacobi?
**Answer:**

Gauss-Seidel converges faster than Jacobi due to two key mechanisms:

**a) Immediate Information Propagation:**
- In Jacobi, the $k$-th iteration uses values from the $(k-1)$-th iteration only:
  $$
  x_i^{(k)} = \frac{1}{a_{ii}}\left(b_i - \sum_{j<i} a_{ij}x_j^{(k-1)} - \sum_{j>i} a_{ij}x_j^{(k-1)}\right)
  $$
- In Gauss-Seidel, we use newly updated values as soon as they're available:
  $$
  x_i^{(k)} = \frac{1}{a_{ii}}\left(b_i - \sum_{j<i} a_{ij}x_j^{(k)} - \sum_{j>i} a_{ij}x_j^{(k-1)}\right)
  $$

**b) Spectral Radius Improvement:**
For a tridiagonal matrix with constant coefficients:
- Jacobi spectral radius: $\rho_J = \cos(\pi h) \approx 1 - \frac{\pi^2 h^2}{2}$
- Gauss-Seidel spectral radius: $\rho_{GS} = \rho_J^2 \approx (1 - \frac{\pi^2 h^2}{2})^2$

This means $\rho_{GS} < \rho_J$, leading to faster convergence.

**c) Physical Interpretation:**
In our heat equation example:
- Jacobi: Temperature updates propagate one cell per iteration
- Gauss-Seidel: Temperature updates propagate multiple cells per iteration, like heat diffusing more naturally

#### 2. What's the preconditioner in PCG and how does it work?
**Answer:**

The preconditioner in PCG is a matrix $M \approx A$ that is easier to invert than $A$. In our implementation:

**a) Diagonal Preconditioner:**
$$
M = \text{diag}(A) = \begin{pmatrix}
a_{11} & 0 & \cdots & 0 \\
0 & a_{22} & \cdots & 0 \\
\vdots & \vdots & \ddots & \vdots \\
0 & 0 & \cdots & a_{nn}
\end{pmatrix}
$$

**b) Why Diagonal?**
- Trivial to invert: $M^{-1}_{ii} = \frac{1}{a_{ii}}$
- Preserves SPD property if $A$ is SPD
- Computational cost: O(N) per application
- Effective when diagonal elements dominate

**c) Preconditioned System:**
The preconditioned system becomes:
$$
M^{-1}Ax = M^{-1}b
$$
This has better conditioning than the original system.

**d) Algorithm Impact:**
The preconditioner modifies the search direction calculation:
```cpp
// Original CG: p_{k+1} = r_{k+1} + β_k p_k
// Preconditioned CG: p_{k+1} = M^{-1}r_{k+1} + β_k p_k
```

**e) Effect on Convergence:**
For our heat equation:
- Unconditioned PCG: ~20 iterations for 1e-6 tolerance
- Preconditioned PCG: ~15 iterations
- Improvement factor depends on problem condition number

#### 3. How would you add a matrix-free solver to our factory?
**Answer:**

Adding a matrix-free solver requires rethinking our interface while maintaining compatibility:

**a) Matrix-Free Concept:**
Instead of storing matrix coefficients, compute matrix-vector products on-the-fly using the underlying physics.

**b) Modified Interface:**
```cpp
template<typename T>
class MatrixFreeSolver : public LinearSolver<T> {
protected:
    // Virtual function for matrix-vector product
    virtual void matrixVectorProduct(const Vector<T>& x, Vector<T>& result) const = 0;

public:
    MatrixFreeSolver(Vector<T>& solution, const Vector<T>& rhs)
        : LinearSolver<T>(/* dummy matrix */, solution, rhs) {}

    bool solve(scalar tolerance, int maxIterations) override;
};
```

**c) Example Implementation:**
```cpp
template<typename T>
class HeatEquationMatrixFree : public MatrixFreeSolver<T> {
private:
    int N_;
    scalar dx_;
    scalar k_;

    void matrixVectorProduct(const Vector<T>& x, Vector<T>& result) const override {
        for (int i = 0; i < N_; ++i) {
            if (i == 0 || i == N_ - 1) {
                result[i] = x[i];  // Boundary conditions
            } else {
                result[i] = -k_/dx_/dx_ * (x[i-1] - 2*x[i] + x[i+1]);
            }
        }
    }

public:
    HeatEquationMatrixFree(Vector<T>& solution, const Vector<T>& rhs, int N, scalar L, scalar k)
        : MatrixFreeSolver<T>(solution, rhs), N_(N), dx_(L/(N+1)), k_(k) {}
};
```

**d) Factory Integration:**
The existing factory can handle matrix-free solvers since they inherit from the same base class:
```cpp
// Register matrix-free solver
REGISTER_SOLVER(MatrixFreeHeatSolver, double);
```

**e) Benefits:**
- Memory efficient: O(N) instead of O(N²) for matrix storage
- Better cache utilization
- Can handle complex geometries without explicit matrix assembly
- Enables adaptive mesh refinement without matrix restructuring

#### 4. Extending to 2D: What changes and what stays the same?
**Answer:**

**a) What Stays the Same:**

- **Linear solver interface**: Same `solve()` method signature

**b) What Changes:**

**Matrix Structure:**
```cpp
// 1D: Tridiagonal
// - Only 3 diagonals
// - Compact storage possible

// 2D: Block tridiagonal with blocks
// - More complex structure
// Typically stored as Compressed Sparse Row (CSR)
```

**Heat Equation in 2D:**
$$
-\frac{\partial}{\partial x}\left(k\frac{\partial T}{\partial x}\right) - \frac{\partial}{\partial y}\left(k\frac{\partial T}{\partial y}\right) = 0
$$

**Matrix Assembly Changes:**
```cpp
// 1D assembly code:
matrix.insert(i, i-1, coeff);     // Left neighbor
matrix.insert(i, i, -2*coeff);    // Center
matrix.insert(i, i+1, coeff);     // Right neighbor

// 2D assembly code (5-point stencil):
matrix.insert(i, i-1, coeff);     // Left neighbor
matrix.insert(i, i+1, coeff);     // Right neighbor
matrix.insert(i, i-Nx, coeff);    // Bottom neighbor
matrix.insert(i, i+Nx, coeff);    // Top neighbor
matrix.insert(i, i, -4*coeff);   // Center
```

**Memory Requirements:**
- **1D**: $O(N)$ storage
- **2D**: $O(N^2)$ storage (but sparse, so $O(N)$ for structured grids)

**Convergence Changes:**
- Jacobi: $\rho_J = \frac{\cos(\pi h_x) + \cos(\pi h_y)}{2}$
- Gauss-Seidel: Still faster than Jacobi, but ratio changes
- Optimal SOR parameter: $\omega_{opt} = \frac{2}{1 + \sqrt{1 - \rho_J^2}}$
