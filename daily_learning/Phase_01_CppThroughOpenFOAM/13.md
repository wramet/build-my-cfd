## Part 3: Implementation - Phase 1

### Step 1: Core Field Class

Create file `mini_field/field.H`:

```cpp
#ifndef mini_field_field_H
#define mini_field_field_H

#include <vector>
#include <iostream>
#include <stdexcept>
#include <algorithm>
#include <cmath>
#include <type_traits>

// -[1] Forward declarations
template<typename T> class tmp;

// -[2] Reference counting base
class refCount
{
private:
    mutable int count_;

public:
    refCount() : count_(0) {}
    virtual ~refCount() = default;

    void operator++() const { count_++; }
    void operator++(int) const { count_++; }
    void operator--() const { count_--; }
    void operator--(int) const { count_--; }

    bool okToDelete() const { return count_ == 0; }
    int count() const { return count_; }
};

// -[3] Main Field class
template<typename T>
class Field : public refCount
{
private:
    std::vector<T> data_;
    std::string name_;

public:
    // -[4] Constructors
    Field() : name_("unnamed") {}
    explicit Field(std::size_t n, const std::string& name = "field")
    : data_(n), name_(name) {}
    Field(std::size_t n, const T& val, const std::string& name = "field")
    : data_(n, val), name_(name) {}

    // -[5] Copy constructor
    Field(const Field& other)
    : data_(other.data_), name_(other.name_ + "_copy")
    {
        std::cout << "    [Field] Copied: " << name_ << "\n";
    }

    // -[6] Move constructor ⭐
    Field(Field&& other) noexcept
    : data_(std::move(other.data_)), name_(other.name_ + "_moved")
    {
        std::cout << "    [Field] Moved: " << name_ << "\n";
    }

    // -[7] Copy assignment
    Field& operator=(const Field& other)
    {
        if (this != &other)
        {
            data_ = other.data_;
            std::cout << "    [Field] Copy-assigned\n";
        }
        return *this;
    }

    // -[8] Move assignment ⭐
    Field& operator=(Field&& other) noexcept
    {
        if (this != &other)
        {
            data_ = std::move(other.data_);
            std::cout << "    [Field] Move-assigned\n";
        }
        return *this;
    }

    // -[9] Destructor
    ~Field()
    {
        std::cout << "    [Field] Destroyed: " << name_ << "\n";
    }

    // -[10] Accessors
    T& operator[](std::size_t i) { return data_[i]; }
    const T& operator[](std::size_t i) const { return data_[i]; }

    std::size_t size() const { return data_.size(); }
    const std::string& name() const { return name_; }

    void setName(const std::string& name) { name_ = name; }

    // -[11] Iterator support
    typedef typename std::vector<T>::iterator iterator;
    typedef typename std::vector<T>::const_iterator const_iterator;

    iterator begin() { return data_.begin(); }
    iterator end() { return data_.end(); }

    const_iterator begin() const { return data_.begin(); }
    const_iterator end() const { return data_.end(); }

    // -[12] Mathematical operations
    T sum() const
    {
        T result = T();
        for (const auto& val : data_) result += val;
        return result;
    }

    T max() const
    {
        if (data_.empty()) return T();
        return *std::max_element(data_.begin(), data_.end());
    }

    T min() const
    {
        if (data_.empty()) return T();
        return *std::min_element(data_.begin(), data_.end());
    }

    T average() const
    {
        if (data_.empty()) return T();
        return sum() / static_cast<T>(data_.size());
    }

    // -[13] Compound assignment operators
    Field& operator+=(const Field& other)
    {
        if (size() != other.size())
            throw std::runtime_error("Size mismatch in +=");
        for (std::size_t i = 0; i < size(); ++i)
            data_[i] += other.data_[i];
        return *this;
    }

    Field& operator-=(const Field& other)
    {
        if (size() != other.size())
            throw std::runtime_error("Size mismatch in -=");
        for (std::size_t i = 0; i < size(); ++i)
            data_[i] -= other.data_[i];
        return *this;
    }

    Field& operator*=(const T& scalar)
    {
        for (auto& val : data_) val *= scalar;
        return *this;
    }

    Field& operator/=(const T& scalar)
    {
        for (auto& val : data_) val /= scalar;
        return *this;
    }

    // -[14] Friend operators
    friend Field operator+(const Field& a, const Field& b)
    {
        if (a.size() != b.size())
            throw std::runtime_error("Size mismatch in +");
        Field result(a.size(), T(), a.name_ + "+" + b.name_);
        for (std::size_t i = 0; i < a.size(); ++i)
            result[i] = a[i] + b[i];
        return result;
    }

    friend Field operator*(const Field& f, const T& scalar)
    {
        Field result(f.size(), T(), f.name_ + "_scaled");
        for (std::size_t i = 0; i < f.size(); ++i)
            result[i] = f[i] * scalar;
        return result;
    }

    friend std::ostream& operator<<(std::ostream& os, const Field& f)
    {
        os << "Field '" << f.name_ << "' [" << f.size() << "]: ";
        for (std::size_t i = 0; i < std::min(std::size_t(5), f.size()); ++i)
            os << f.data_[i] << " ";
        if (f.size() > 5) os << "...";
        return os;
    }
};

// -[15] Convenience typedefs
typedef Field<double> scalarField;
typedef Field<int> labelField;

#endif
```

### Step 2: Smart Pointer (tmp)

Create file `mini_field/tmp.H`:

```cpp
#ifndef mini_field_tmp_H
#define mini_field_tmp_H

#include "field.H"

// -[1] Smart pointer for Field management
template<typename T>
class tmp : public refCount
{
private:
    T* ptr_;
    mutable bool isRef_;

public:
    // -[2] Constructors
    tmp() : ptr_(nullptr), isRef_(false) {}

    explicit tmp(T* p) : ptr_(p), isRef_(false)
    {
        if (ptr_) ptr_->operator++();
    }

    tmp(const tmp& t) : ptr_(t.ptr_), isRef_(true)
    {
        if (ptr_) ptr_->operator++();
    }

    // -[3] Move constructor
    tmp(tmp&& t) noexcept : ptr_(t.ptr_), isRef_(t.isRef_)
    {
        t.ptr_ = nullptr;
    }

    // -[4] Destructor
    ~tmp()
    {
        if (ptr_)
        {
            ptr_->operator--();
            if (ptr_->okToDelete())
            {
                delete ptr_;
            }
        }
    }

    // -[5] Access
    T& operator*() { return *ptr_; }
    T* operator->() { return ptr_; }
    T* ptr() { return ptr_; }

    const T& operator*() const { return *ptr_; }
    const T* operator->() const { return ptr_; }
    const T* ptr() const { return ptr_; }

    // -[6] Reference management
    void ref()
    {
        if (ptr_)
        {
            isRef_ = true;
            ptr_->operator++();
        }
    }

    T* release()
    {
        T* tmp = ptr_;
        ptr_ = nullptr;
        return tmp;
    }

    void clear()
    {
        if (ptr_ && !isRef_)
        {
            delete ptr_;
        }
        ptr_ = nullptr;
    }
};

#endif
```

## Part 4: Testing - Phase 1

Create file `mini_field/test_phase1.C`:

```cpp
#include "field.H"
#include "tmp.H"
#include <iostream>

int main()
{
    std::cout << "=== Mini-Project Part 1: Core Field<T> ===\n\n";

    // -[1] Basic usage
    std::cout << "Test 1: Basic Field Operations\n";
    scalarField p(5, 101325.0, "pressure");
    std::cout << p << "\n";

    // -[2] Mathematical operations
    std::cout << "\nTest 2: Math Operations\n";
    std::cout << "  Max: " << p.max() << "\n";
    std::cout << "  Min: " << p.min() << "\n";
    std::cout << "  Sum: " << p.sum() << "\n";
    std::cout << "  Average: " << p.average() << "\n";

    // -[3] Compound assignment
    std::cout << "\nTest 3: Compound Assignment\n";
    scalarField deltaP(5, 1000.0, "deltaP");
    p += deltaP;
    std::cout << "  After +=: " << p[0] << "\n";

    p *= 1.1;
    std::cout << "  After *= 1.1: " << p[0] << "\n";

    // -[4] Move semantics
    std::cout << "\nTest 4: Move Semantics\n";
    scalarField moved = std::move(p);
    std::cout << "  Moved field: " << moved[0] << "\n";

    // -[5] Iterators
    std::cout << "\nTest 5: Iterator Support\n";
    scalarField T(5, 300.0, "temperature");
    for (auto& val : T)
    {
        val += 10.0;
    }
    std::cout << "  After iteration: " << T[0] << "\n";

    // -[6] Smart pointer
    std::cout << "\nTest 6: tmp<> Smart Pointer\n";
    tmp<scalarField> tRho(new scalarField(5, 1.2, "density"));
    tRho->operator[](0) = 1.225;
    std::cout << "  tmp field value: " << (*tRho)[0] << "\n";

    std::cout << "\n✅ Part 1 Complete: Core Field<T> implementation!\n";
    return 0;
}
```
