# Day 22: HashTable<T> - Open Addressing in OpenFOAM

## Introduction to HashTable Design (การแนะนำ HashTable)

OpenFOAM's `HashTable<T>` implements open addressing for hash table storage, contrasting with STL's `std::unordered_map` which uses chaining. This design choice provides better cache performance for typical CFD workloads where hash tables are frequently accessed in tight loops.

```cpp
// Example HashTable usage in OpenFOAM
HashTable<scalar> fieldProperties;
fieldProperties.insert("density", 1.2);
fieldProperties.insert("viscosity", 0.001);

scalar rho = fieldProperties["density"];  // Direct access
```

Open addressing stores all elements in a single array, with collisions resolved by probing for the next available slot. This eliminates pointer chasing and provides better spatial locality.

## Open Addressing Fundamentals

⭐ ### Design Principles (หลักการออกแบบ)

Open addressing follows these core principles:
1. **Single storage array**: All elements stored contiguously
2. **Probe sequence**: Deterministic sequence of slots to check
3. **Tombstone mechanism**: Special marker for deleted elements
4. **Load factor control**: Dynamic resizing to maintain performance

⭐ ### Mathematical Foundation (พื้นฐานทางคณิตศาสตร์)

Given a hash table with capacity $M$ and hash function $h$, the probe sequence is:

$$
p(k, i) = (h(k) + c_1 i + c_2 i^2) \bmod M
$$

Where:
- $i$ is the probe number
- $c_1$ and $c_2$ are probe sequence constants
- For linear probing: $c_1 = 1$, $c_2 = 0$
- For quadratic probing: $c_1 = c_2 = 1$

The load factor $\alpha$ is:

$$
\alpha = \frac{n}{M}
$$

Where $n$ is the number of elements.

## HashTable<T> Implementation (การเขียน HashTable<T>)

```cpp
// File: /Users/woramet/Documents/Build My CFD/.claude/worktrees/upbeat-banach/daily_learning/Phase_02_DataStructures_Memory/22.md
// HashTable<T> with open addressing implementation

template<class Key, class T>
class HashTable
{
private:
    // Hash table entry
    struct Entry
    {
        Key key;
        T value;
        bool occupied;
        bool deleted;

        Entry() : occupied(false), deleted(false) {}
    };

    // Storage array
    DynamicList<Entry> table_;

    // Hash function
    size_t hash(const Key& key) const {
        // Simple hash function - in OpenFOAM this is more sophisticated
        return std::hash<Key>{}(key);
    }

    // Probe sequence: linear probing
    size_t probe(size_t hashVal, size_t i, size_t capacity) const {
        return (hashVal + i) % capacity;
    }

    // Find entry in table
    Entry* findEntry(const Key& key) {
        if (empty()) return nullptr;

        size_t hashVal = hash(key) % table_.capacity();
        size_t firstDeleted = table_.capacity();
        size_t probeCount = 0;

        while (probeCount < table_.capacity()) {
            Entry& entry = table_[probeCount];
            size_t pos = probe(hashVal, probeCount, table_.capacity());

            if (!entry.occupied) {
                // Empty slot - not found
                return nullptr;
            }

            if (entry.deleted) {
                if (firstDeleted == table_.capacity()) {
                    firstDeleted = pos;
                }
            } else if (entry.key == key) {
                // Found
                return &entry;
            }

            probeCount++;
        }

        return nullptr;
    }

public:
    // Constructor: HashTable(size, value, "name")
    HashTable() {}

    explicit HashTable(const label size) : table_(size) {}

    bool insert(const Key& key, const T& value) {
        if (loadFactor() > 0.7) {
            resize(table_.capacity() * 2);
        }

        size_t hashVal = hash(key) % table_.capacity();
        size_t firstDeleted = table_.capacity();

        for (size_t i = 0; i < table_.capacity(); ++i) {
            size_t pos = probe(hashVal, i, table_.capacity());
            Entry& entry = table_[pos];

            if (!entry.occupied) {
                // Found empty slot - insert here
                entry.key = key;
                entry.value = value;
                entry.occupied = true;
                entry.deleted = false;
                return true;
            }

            if (entry.deleted) {
                if (firstDeleted == table_.capacity()) {
                    firstDeleted = pos;
                }
            } else if (entry.key == key) {
                // Key already exists - update
                entry.value = value;
                return true;
            }
        }

        // Table is full - should have been resized earlier
        return false;
    }

    bool erase(const Key& key) {
        Entry* entry = findEntry(key);
        if (entry) {
            entry->deleted = true;
            return true;
        }
        return false;
    }

    T& operator[](const Key& key) {
        Entry* entry = findEntry(key);
        if (entry) {
            return entry->value;
        }

        // Insert new element
        insert(key, T());
        return findEntry(key)->value;
    }

    const T& operator[](const Key& key) const {
        Entry* entry = findEntry(key);
        if (!entry) {
            throw std::out_of_range("Key not found");
        }
        return entry->value;
    }

    bool contains(const Key& key) const {
        return findEntry(key) != nullptr;
    }

    bool empty() const {
        return table_.empty();
    }

    size_t size() const {
        size_t count = 0;
        for (label i = 0; i < table_.size(); ++i) {
            if (table_[i].occupied && !table_[i].deleted) {
                count++;
            }
        }
        return count;
    }

    size_t capacity() const {
        return table_.capacity();
    }

    double loadFactor() const {
        return static_cast<double>(size()) / capacity();
    }

private:
    void resize(size_t newSize) {
        DynamicList<Entry> oldTable = table_;
        table_.setSize(newSize);

        // Reinsert all active entries
        for (label i = 0; i < oldTable.size(); ++i) {
            Entry& entry = oldTable[i];
            if (entry.occupied && !entry.deleted) {
                insert(entry.key, entry.value);
            }
        }
    }
};
```

```cpp
// Power-of-2 sizing

size_t nextPowerOfTwo(size_t n) {
    if (n == 0) return 1;
    --n;
    n |= n >> 1;
    n |= n >> 2;
    n |= n >> 4;
    n |= n >> 8;
    n |= n >> 16;
    n |= n >> 32;
    return ++n;
}

// Hash table with power-of-2 capacity
template<class Key, class T>
class PowerOfTwoHashTable : public HashTable<Key, T>
{
public:
    PowerOfTwoHashTable() : HashTable<Key, T>() {}

    explicit PowerOfTwoHashTable(size_t initialSize)
    : HashTable<Key, T>(nextPowerOfTwo(initialSize)) {}

    // Optimized probe using bit masking
    size_t probe(size_t hashVal, size_t i, size_t capacity) const {
        // For power-of-2, modulo becomes bitwise AND
        return (hashVal + i) & (capacity - 1);
    }
};
```

## Comparison with std::unordered_map (เปรียบเทียบกับ std::unordered_map)

⭐ ### Implementation Differences (ความแตกต่างในการ implement)

| Feature | OpenFOAM HashTable | std::unordered_map |
|---------|-------------------|-------------------|
| Collision resolution | Linear probing | Chaining (separate lists) |
| Storage | Single contiguous array | Array + separate arrays |
| Cache performance | Excellent | Good (pointer chasing) |
| Memory overhead | Minimal | Higher (node allocations) |
| Deletion | Tombstones | Free list |
| Performance characteristics | Better for small-to-medium tables | Better for large tables |

### Cache Performance Analysis

The cache miss rate differs significantly between implementations:

#### OpenFOAM HashTable (Open Addressing)
$$
P_{\text{miss}} = \frac{\text{probe length} \times \text{entry size}}{\text{cache line size}}
$$

For average probe length of 2-3 entries and 64B cache line:
- 32B entry size: ~100% hit rate
- 64B entry size: ~50% hit rate

#### std::unordered_map (Chaining)
$$
P_{\text{miss}} \approx 1 - \frac{1}{1 + \frac{\text{cache line size}}{\text{node size + pointer size}}}
$$

For 64B cache line, 24B node + 8B pointer:
- ~89% miss rate

```cpp
            pointIndex_[std::to_string(pointLabels[i])] = i;
        }
    }

    label findCell(const label cellId) const {

        DynamicList<label> indices;
        indices.reserve(cellIds.size());

        for (label id : cellIds) {
            label idx = findCell(id);
            indices.append(idx);
        }
        return indices;
    }
```

## Complete HashTable Implementation with Benchmarking

```cpp
// File: /Users/woramet/Documents/Build My CFD/.claude/worktrees/upbeat-banach/daily_learning/Phase_02_DataStructures_Memory/22.md
// Complete HashTable implementation

#ifndef HASHTABLE_H
#define HASHTABLE_H

#include <functional>
#include <chrono>
#include <random>

// Hash table entry
template<class Key, class T>
struct HashEntry
{
    Key key;
    T value;
    bool occupied;
    bool deleted;

    HashEntry() : occupied(false), deleted(false) {}

    bool isActive() const {
        return occupied && !deleted;
    }
};

// HashTable with linear probing
template<class Key, class T>
class HashTable
{
private:
    DynamicList<HashEntry<Key, T>> table_;
    size_t size_;
    double maxLoadFactor_;

    // Hash functions
    size_t hashKey(const Key& key) const {
        return std::hash<Key>{}(key);
    }

    size_t probeSequence(size_t hashVal, size_t i, size_t capacity) const {
        return (hashVal + i) % capacity;
    }

    // Find entry
    HashEntry<Key, T>* findEntry(const Key& key) {
        if (size_ == 0) return nullptr;

        size_t hashVal = hashKey(key);
        for (size_t i = 0; i < table_.capacity(); ++i) {
            size_t pos = probeSequence(hashVal, i, table_.capacity());
            auto& entry = table_[pos];

            if (!entry.occupied) {
                return nullptr;  // Empty slot - not found
            }

            if (entry.isActive() && entry.key == key) {
                return &entry;  // Found
            }
        }

        return nullptr;  // Wrapped around - not found
    }

    // Resize table
    void resize(size_t newSize) {
        DynamicList<HashEntry<Key, T>> oldTable = table_;
        table_.setSize(newSize);

        // Clear new table
        for (size_t i = 0; i < table_.size(); ++i) {
            table_[i] = HashEntry<Key, T>();
        }

        // Reinsert active entries
        for (size_t i = 0; i < oldTable.size(); ++i) {
            auto& entry = oldTable[i];
            if (entry.isActive()) {
                insert(entry.key, entry.value);
            }
        }
    }

public:
    // Constructor: HashTable(size, value, "name")
    HashTable() : size_(0), maxLoadFactor_(0.7) {}

    explicit HashTable(size_t capacity)
    : table_(capacity), size_(0), maxLoadFactor_(0.7) {}

    // Insert or update
    bool insert(const Key& key, const T& value) {
        // Check if key exists
        auto* existing = findEntry(key);
        if (existing) {
            existing->value = value;  // Update existing
            return true;
        }

        // Resize if needed
        if (static_cast<double>(size_) / table_.capacity() >= maxLoadFactor_) {
            resize(table_.capacity() == 0 ? 8 : table_.capacity() * 2);
        }

        // Find empty slot for insertion
        size_t hashVal = hashKey(key);
        for (size_t i = 0; i < table_.capacity(); ++i) {
            size_t pos = probeSequence(hashVal, i, table_.capacity());
            auto& entry = table_[pos];

            if (!entry.occupied || entry.deleted) {
                // Insert here
                entry.key = key;
                entry.value = value;
                entry.occupied = true;
                entry.deleted = false;
                size_++;
                return true;
            }
        }

        return false;  // Shouldn't happen with proper resizing
    }

    // Erase
    bool erase(const Key& key) {
        auto* entry = findEntry(key);
        if (entry) {
            entry->deleted = true;
            size_--;
            return true;
        }
        return false;
    }

    // Access
    T& operator[](const Key& key) {
        auto* entry = findEntry(key);
        if (entry) {
            return entry->value;
        }

        // Insert new element
        insert(key, T());
        return findEntry(key)->value;
    }

    const T& operator[](const Key& key) const {
        auto* entry = findEntry(key);
        if (!entry) {
            throw std::out_of_range("Key not found");
        }
        return entry->value;
    }

    // Contains
    bool contains(const Key& key) const {
        return findEntry(key) != nullptr;
    }

    // Size operations
    size_t size() const { return size_; }
    bool empty() const { return size_ == 0; }
    size_t capacity() const { return table_.capacity(); }

    // Clear
    void clear() {
        for (size_t i = 0; i < table_.size(); ++i) {
            table_[i] = HashEntry<Key, T>();
        }
        size_ = 0;
    }

    // Set max load factor
    void setMaxLoadFactor(double factor) {
        maxLoadFactor_ = factor;
        if (static_cast<double>(size_) / table_.capacity() > factor) {
            resize(table_.capacity() * 2);
        }
    }
};

    }
```

## Mathematical Analysis of Performance

### Expected Probe Length

For a hash table with load factor $\alpha$, the expected probe length for successful search is:

$$
E[\text{probe length}] = \frac{1}{2} \left(1 + \frac{1}{1 - \alpha}\right)
$$

For unsuccessful search:

$$
E[\text{probe length}] = \frac{1}{2} \left(1 + \frac{1}{(1 - \alpha)^2}\right)
$$

At $\alpha = 0.7$:
- Successful search: ~1.67 probes
- Unsuccessful search: ~2.89 probes

### Space Complexity

The space complexity is:

$$
S = n \cdot (\text{sizeof(Key)} + \text{sizeof(T)} + \text{overhead}) + M \cdot \text{sizeof(empty\_entry})
$$

Where $M$ is the table size and $n$ is the number of elements.

### Time Complexity

| Operation | Average Case | Worst Case |
|-----------|-------------|------------|
| Insert | $O(1)$ | $O(n)$ |
| Delete | $O(1)$ | $O(n)$ |
| Search | $O(1)$ | $O(n)$ |
| Resize | $O(n)$ | $O(n)$ |

```cpp
// Solution
template<class Key, class T>
class HashTableIterator
{
private:
    HashTable<Key, T>& table_;
    size_t currentIndex_;
    size_t activeCount_;

public:
    HashTableIterator(HashTable<Key, T>& table, size_t index = 0)
    : table_(table), currentIndex_(index), activeCount_(0) {}

    bool operator!=(const HashTableIterator& other) const {
        return currentIndex_ != other.currentIndex_;
    }

    HashTableIterator& operator++() {
        do {
            currentIndex_++;
        } while (currentIndex_ < table_.capacity() &&
                !table_.table_[currentIndex_].isActive());

        return *this;
    }

    auto& operator*() {
        return table_.table_[currentIndex_];
    }
};

// HashTable with iterator support
template<class Key, class T>
class IteratorHashTable : public HashTable<Key, T>
{
public:
    using iterator = HashTableIterator<Key, T>;

    iterator begin() {
        size_t start = 0;
        while (start < this->capacity() &&
               !this->table_[start].isActive()) {
            start++;
        }
        return iterator(*this, start);
    }

    iterator end() {
        return iterator(*this, this->capacity());
    }
```

### Exercise 6: HashTable with Collision Statistics
Implement statistics tracking for hash table collisions:

```cpp
// Solution
template<class Key, class T>
class StatisticsHashTable : public HashTable<Key, T>
{
private:
    std::vector<size_t> probeLengths_;
    size_t maxProbeLength_;

public:
    StatisticsHashTable() : maxProbeLength_(0) {}

    bool insert(const Key& key, const T& value) override {
        size_t probeLength = 0;

        // Find insertion point and track probe length
        size_t hashVal = this->hashKey(key);
        for (size_t i = 0; i < this->table_.capacity(); ++i) {
            size_t pos = this->probeSequence(hashVal, i, this->table_.capacity());
            auto& entry = this->table_[pos];

            if (!entry.occupied || entry.deleted) {
                // Found insertion point
                entry.key = key;
                entry.value = value;
                entry.occupied = true;
                entry.deleted = false;
                this->size_++;

                // Track probe length
                probeLengths_.push_back(probeLength + 1);
                maxProbeLength_ = Foam::max(maxProbeLength_, probeLength + 1);

                return true;
            }

            probeLength++;
        }

        return false;
    }

    // Get statistics
    size_t getMaxProbeLength() const { return maxProbeLength_; }
    double getAverageProbeLength() const {
        if (probeLengths_.empty()) return 0.0;
        double sum = 0.0;
        for (size_t length : probeLengths_) {
            sum += length;
        }
        return sum / probeLengths_.size();
    }

    void printStatistics() const {
        std::cout << "Hash Table Statistics:" << std::endl;
        std::cout << "  Size: " << this->size() << std::endl;
        std::cout << "  Capacity: " << this->capacity() << std::endl;
        std::cout << "  Load Factor: " << this->loadFactor() << std::endl;
        std::cout << "  Max Probe Length: " << maxProbeLength_ << std::endl;
        std::cout << "  Average Probe Length: " << getAverageProbeLength() << std::endl;
    }
};
```
