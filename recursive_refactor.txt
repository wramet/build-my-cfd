import os
import subprocess
import sys
import re
from pathlib import Path

# ==========================================
# ⚙️ CONFIGURATION
# ==========================================

TARGET_MODULE_ROOT = "/Users/woramet/Documents/th_new/MODULE_04_MULTIPHASE_FUNDAMENTALS"
LOG_FILE = "completed_recursive.log"

CLI_COMMAND = "claude"
CLI_FLAGS = ["--dangerously-skip-permissions", "-p"]

# จำกัดขนาด Context (Characters) เพื่อกัน Token Overflow
# 1 Token ≈ 4 chars. Limit 128k tokens ≈ 500k chars. 
# ตั้งไว้ 300,000 chars เพื่อความปลอดภัย (เผื่อ Prompt และ Output)
MAX_CONTEXT_CHARS = 300000 

# ==========================================

def get_smart_prompt(target_filename, relevant_files_list):
    """
    Prompt ที่บอก AI ว่าเราคัดไฟล์มาให้แล้ว
    """
    files_str = ", ".join(relevant_files_list)
    return f"""
**Role:** Senior Technical Editor & OpenFOAM Subject Matter Expert.

**Context:**
- **Target File:** `{target_filename}` (Draft).
- **Selected Reference Files:** `{files_str}` (Most relevant Backups).

**Task:** High-Fidelity Content Restoration.

**Instructions:**
1. **Search & Restore:** I have analyzed the text and selected the most relevant `.bak` files for you. Search inside these files to find technical details (equations, code, tables) matching the topics in `{target_filename}`.
2. **Action:** Overwrite `{target_filename}` with enriched content.
   - **MUST RESTORE:** LaTeX equations, OpenFOAM code snippets, parameter details.
3. **Language:** **Thai** (Preserve English technical terms).
4. **Constraint:** Edit ONLY `{target_filename}`. Do not summarize.

**Output:** Full enriched content for `{target_filename}`.
"""

def extract_keywords(text):
    """แยกคำออกมาเป็น Set เพื่อใช้เทียบความเหมือน"""
    # เอาเฉพาะตัวอักษรภาษาอังกฤษและไทย ตัดคำง่ายๆ
    words = re.findall(r'\w+', text.lower())
    # กรองคำสั้นๆ ออก (Stop words แบบบ้านๆ)
    return set(w for w in words if len(w) > 3)

def select_relevant_backups(md_file_path, bak_files, folder_path):
    """
    คัดเลือกไฟล์ .bak ที่มีความเกี่ยวข้องกับ .md มากที่สุด จนกว่าจะเต็มโควต้า
    """
    try:
        with open(md_file_path, 'r', encoding='utf-8') as f:
            md_content = f.read()
            md_keywords = extract_keywords(md_content)
    except Exception:
        return bak_files # ถ้าอ่านไม่ได้ ให้ส่งไปหมดเลย (Fallback)

    scored_files = []
    
    for bak_file in bak_files:
        bak_path = Path(folder_path) / bak_file
        try:
            with open(bak_path, 'r', encoding='utf-8') as f:
                content = f.read()
                # 1. คำนวณความเหมือน (Jaccard Similarity แบบง่าย)
                bak_keywords = extract_keywords(content)
                common_words = md_keywords.intersection(bak_keywords)
                score = len(common_words)
                
                # 2. เก็บข้อมูลขนาดไฟล์
                size = len(content)
                scored_files.append({
                    'name': bak_file,
                    'score': score,
                    'size': size
                })
        except Exception:
            continue

    # เรียงลำดับตามคะแนน (มากไปน้อย)
    scored_files.sort(key=lambda x: x['score'], reverse=True)

    selected_files = []
    current_size = 0
    
    # วนลูปหยิบไฟล์ใส่กระเป๋า จนกว่าจะเต็ม
    for file_data in scored_files:
        if current_size + file_data['size'] < MAX_CONTEXT_CHARS:
            selected_files.append(file_data['name'])
            current_size += file_data['size']
        else:
            # ถ้าเต็มแล้ว หยุดหยิบ (ไฟล์ที่เหลือคือคะแนนน้อยกว่า ไม่จำเป็นต้องเอาไป)
            break
            
    return selected_files

def create_ignore_file(folder_path):
    ignore_path = Path(folder_path) / ".claudeignore"
    content = "*.png\n*.jpg\n*.jpeg\n*.pdf\n*.zip\n.DS_Store\n.git\n"
    try:
        with open(ignore_path, "w") as f:
            f.write(content)
    except Exception:
        pass 

def main():
    try:
        root_path = Path(TARGET_MODULE_ROOT).resolve()
        log_path = Path(LOG_FILE).resolve()
    except Exception:
        sys.exit(1)

    print(f"🚀 Starting Smart Refactoring (Token Optimized)")
    print(f"📂 Target: {root_path}")
    
    if not root_path.exists():
        sys.exit(1)
        
    if not log_path.exists():
        log_path.touch()

    with open(log_path, 'r', encoding='utf-8') as f:
        completed_paths = set(line.strip() for line in f if line.strip())

    for current_root, dirs, files in os.walk(root_path):
        
        md_files = sorted([f for f in files if f.endswith('.md')])
        bak_files = sorted([f for f in files if f.endswith('.bak')])

        if not md_files or not bak_files:
            continue

        abs_current_path = str(Path(current_root).resolve())
        folder_name = Path(current_root).name

        if abs_current_path in completed_paths:
            continue

        print(f"\n🔨 Processing Folder: {folder_name}")
        create_ignore_file(abs_current_path)

        all_success = True
        
        for md_file in md_files:
            print(f"   👉 Analyzing: {md_file} ...")
            
            # 🔥 SMART SELECTION: คัดเลือกไฟล์ .bak ที่เกี่ยวข้องที่สุด
            relevant_bak_files = select_relevant_backups(
                Path(abs_current_path) / md_file, 
                bak_files, 
                abs_current_path
            )
            
            print(f"      🧠 Selected {len(relevant_bak_files)}/{len(bak_files)} relevant backups based on content.")
            
            # รวมไฟล์ที่จะส่ง (MD เป้าหมาย + BAK ที่คัดมาแล้ว)
            current_batch = [md_file] + relevant_bak_files
            
            # สร้าง Prompt
            final_prompt = get_smart_prompt(md_file, relevant_bak_files)
            
            cmd = [CLI_COMMAND] + CLI_FLAGS + [final_prompt] + current_batch

            try:
                result = subprocess.run(cmd, text=True, cwd=abs_current_path, capture_output=True)
                if result.returncode == 0:
                    print(result.stdout)
                    print(f"      ✅ Done.")
                else:
                    print(f"      ❌ Failed: {result.stderr}")
                    all_success = False
            except Exception as e:
                print(f"      ❌ Error: {e}")
                all_success = False

        if all_success:
            with open(log_path, 'a', encoding='utf-8') as f:
                f.write(abs_current_path + "\n")

    print("\n🏁 Script Finished.")

if __name__ == "__main__":
    main()
