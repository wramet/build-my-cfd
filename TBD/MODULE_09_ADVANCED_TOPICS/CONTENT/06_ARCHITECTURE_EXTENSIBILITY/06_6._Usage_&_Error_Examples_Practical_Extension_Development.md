# 6. การใช้งานและตัวอย่างข้อผิดพลาด: การพัฒนา Extension ในทางปฏิบัติ

ส่วนนี้ให้ตัวอย่างการใช้งานจริงในการสร้างส่วนขยายแบบกำหนดเองของ OpenFOAM โดยเน้นไปที่การพัฒนา functionObjects ตัวอย่างเหล่านี้ครอบคลุมขั้นตอนการพัฒนาที่สมบูรณ์ตั้งแต่การ implement ไปจนถึงการติดตั้งและการแก้ไขปัญหา

## 6.1 การสร้าง FunctionObject แบบกำหนดเอง

FunctionObjects ใน OpenFOAM ให้กลไกที่ทรงพลังในการขยายความสามารถของ solver โดยไม่ต้องแก้ไขโค้ดหลักของ solver พวกมันช่วยให้สามารถจัดการ field แบบกำหนดเอง, การวิเคราะห์, และการสร้าง output ได้ในระหว่างการทำงานของ simulation

### 6.1.1 ขั้นตอนการพัฒนาที่สมบูรณ์

**ขั้นที่ 1: การกำหนด Interface (`myFunctionObject.H`)**

ไฟล์ header กำหนด interface ของคลาสและสร้างข้อตกลงสำหรับพฤติกรรมของ function object:

```cpp
#ifndef myFunctionObject_H
#define myFunctionObject_H

#include "functionObject.H"
#include "volFields.H"
#include "dictionary.H"
#include "runTimeSelectionTables.H"

namespace Foam
{

class myFunctionObject
:
    public functionObject
{
private:
    // Private data members
    word fieldName_;        // ชื่อของ field ที่จะประมวลผล
    scalar threshold_;      // ค่า threshold สำหรับการกรอง
    const fvMesh& mesh_;    // Reference ไปยัง computational mesh
    
    // Results storage
    autoPtr<volScalarField> filteredField_;

public:
    // Runtime type information
    TypeName("myFilter");

    // Runtime selection support
    declareRunTimeSelectionTable
    (
        autoPtr,
        myFunctionObject,
        dictionary,
        (
            const word& name,
            const Time& runTime,
            const dictionary& dict
        ),
        (name, runTime, dict)
    );

    // Constructors
    myFunctionObject
    (
        const word& name,
        const Time& runTime,
        const dictionary& dict
    );

    // Destructor
    virtual ~myFunctionObject();

    // Member functions inherited from functionObject
    virtual bool execute();
    virtual bool write();
    virtual bool read(const dictionary& dict);
    virtual wordList fields() const;
    virtual void updateMesh(const mapPolyMesh& mpm);
    virtual void movePoints(const polyMesh& mesh);

private:
    // Private helper methods
    void initializeField();
    void applyFilter();
};

} // namespace Foam

#endif
```

**รูปแบบการออกแบบที่สำคัญ:**
- **RAII (Resource Acquisition Is Initialization)**: การจัดการ field อัตโนมัติผ่าน `autoPtr`
- **Template-based field access**: รองรับประเภท field หลายประเภทผ่าน templating
- **Mesh-aware design**: รักษา reference ไปยัง `fvMesh` สำหรับการดำเนินการ field

**ขั้นที่ 2: การ Implement (`myFunctionObject.C`)**

ไฟล์ implementation ให้ฟังก์ชันการทำงานที่จริง:

```cpp
#include "myFunctionObject.H"
#include "addToRunTimeSelectionTable.H"
#include "volFields.H"
#include "dictionaryEntry.H"

// Register with runtime selection system
namespace Foam
{
    defineTypeNameAndDebug(myFunctionObject, 0);
    addToRunTimeSelectionTable
    (
        functionObject,
        myFunctionObject,
        dictionary
    );
}

// Constructor with robust error handling
Foam::myFunctionObject::myFunctionObject
(
    const word& name,
    const Time& runTime,
    const dictionary& dict
)
:
    functionObject(name),
    mesh_(runTime.lookupObject<fvMesh>(dynamic_cast<const fvMesh&>(runTime))),
    fieldName_(word::null),
    threshold_(0.0),
    filteredField_(nullptr)
{
    // Read configuration with error checking
    if (!dict.readIfPresent("fieldName", fieldName_))
    {
        FatalIOErrorIn
        (
            "myFunctionObject::myFunctionObject"
            "(const word&, const Time&, const dictionary&)",
            dict
        ) << "Missing mandatory entry 'fieldName'" << exit(FatalIOError);
    }
    
    if (!dict.readIfPresent("threshold", threshold_))
    {
        WarningIn
        (
            "myFunctionObject::myFunctionObject"
            "(const word&, const Time&, const dictionary&)"
        ) << "Using default threshold value: " << threshold_ << endl;
    }
    
    // Initialize filtered field
    initializeField();
    
    Info << "Created " << type() << " for field " << fieldName_ 
         << " with threshold " << threshold_ << endl;
}

// Destructor
Foam::myFunctionObject::~myFunctionObject()
{
    Info << "Destroying " << type() << endl;
}

// Initialize the filtered field
void Foam::myFunctionObject::initializeField()
{
    if (filteredField_.valid())
    {
        // Field already exists
        return;
    }
    
    // Check if source field exists
    if (!mesh_.foundObject<volScalarField>(fieldName_))
    {
        FatalErrorIn("myFunctionObject::initializeField()")
            << "Field " << fieldName_ << " not found in mesh" 
            << exit(FatalError);
    }
    
    const volScalarField& sourceField = 
        mesh_.lookupObject<volScalarField>(fieldName_);
    
    filteredField_.set
    (
        new volScalarField
        (
            IOobject
            (
                "filtered" + fieldName_,
                mesh_.time().timeName(),
                mesh_,
                IOobject::NO_READ,
                IOobject::AUTO_WRITE
            ),
            mesh_,
            dimensionedScalar("zero", sourceField.dimensions(), 0.0)
        )
    );
}

// Execute the filtering operation
bool Foam::myFunctionObject::execute()
{
    if (!filteredField_.valid())
    {
        WarningIn("myFunctionObject::execute()")
            << "Filtered field not initialized" << endl;
        return false;
    }
    
    try
    {
        applyFilter();
        return true;
    }
    catch (const std::exception& e)
    {
        ErrorIn("myFunctionObject::execute()")
            << "Exception during filtering: " << e.what() << endl;
        return false;
    }
}

// Apply filtering algorithm
void Foam::myFunctionObject::applyFilter()
{
    if (!mesh_.foundObject<volScalarField>(fieldName_))
    {
        FatalErrorIn("myFunctionObject::applyFilter()")
            << "Field " << fieldName_ << " not found" << exit(FatalError);
    }
    
    const volScalarField& sourceField = 
        mesh_.lookupObject<volScalarField>(fieldName_);
    volScalarField& filteredField = filteredField_();
    
    // Vectorized filtering operation
    filteredField = pos(sourceField - threshold_) * sourceField;
    
    // Apply boundary conditions
    filteredField.correctBoundaryConditions();
    
    if (debug)
    {
        scalar minVal = gMin(filteredField.primitiveField());
        scalar maxVal = gMax(filteredField.primitiveField());
        scalar sumVal = gSum(filteredField.primitiveField());
        
        Info << "Filter statistics - min: " << minVal 
             << ", max: " << maxVal << ", sum: " << sumVal << endl;
    }
}

// Write results to disk
bool Foam::myFunctionObject::write()
{
    if (!filteredField_.valid())
    {
        WarningIn("myFunctionObject::write()")
            << "No filtered field to write" << endl;
        return false;
    }
    
    filteredField_().write();
    
    // Write additional statistics
    OFstream statsFile
    (
        mesh_.time().timePath()/"filterStats_" + name() + ".dat"
    );
    
    const volScalarField& filteredField = filteredField_();
    scalar minVal = gMin(filteredField.primitiveField());
    scalar maxVal = gMax(filteredField.primitiveField());
    scalar sumVal = gSum(filteredField.primitiveField());
    
    statsFile << "# Time  Min  Max  Sum" << nl
              << mesh_.time().timeName() << " " 
              << minVal << " " << maxVal << " " << sumVal << nl;
    
    return true;
}

// Return list of fields this functionObject depends on
Foam::wordList Foam::myFunctionObject::fields() const
{
    return wordList(1, fieldName_);
}

// Handle mesh topology changes
void Foam::myFunctionObject::updateMesh(const mapPolyMesh& mpm)
{
    if (filteredField_.valid())
    {
        filteredField_().mapFields(mpm);
    }
}

// Handle mesh point motion
void Foam::myFunctionObject::movePoints(const polyMesh& mesh)
{
    if (filteredField_.valid())
    {
        filteredField_().movePoints();
    }
}

// Re-read configuration
bool Foam::myFunctionObject::read(const dictionary& dict)
{
    dict.readIfPresent("fieldName", fieldName_);
    dict.readIfPresent("threshold", threshold_);
    
    // Reinitialize if field name changed
    initializeField();
    
    return true;
}
```

## 6.2 การจัดการข้อผิดพลาดและการ Debugging ขั้นสูง

### 6.2.1 ข้อผิดพลาดในการ Compile ที่พบบ่อย

**Template Instantiation Errors:**
```cpp
// Error: template instantiation failure
// Solution: Ensure all required template specializations are included
#include "volFieldsFwd.H"  // Forward declarations
#include "volScalarField.H" // Concrete implementations
```

**Linking Errors:**
```bash
# Error: undefined reference to typeinfo
# Solution: Ensure virtual functions are properly defined
virtual ~myFunctionObject() {}  // Always provide virtual destructor
```

### 6.2.2 การแก้ไข Runtime Error

**Dynamic Library Loading Issues:**

```cpp
// Enhanced error checking in constructor
myFunctionObject::myFunctionObject(...)
{
    // Verify mesh object exists

        meshPtr = &runTime.lookupObject<fvMesh>("region0");
```

## 6.3 การรวมเข้ากับ Solvers

### 6.3.1 ตัวอย่างการกำหนดค่ากรณี

การกำหนดค่า `system/controlDict` ที่สมบูรณ์:

```cpp
functions
{
    velocityFilter
    {
        type            myFilter;
        libs            ("libmyFunctionObjects.so");
        
        // Configuration parameters
        fieldName       U;
        threshold       5.0;
        
        // Execution control
        executeControl  timeStep;
        executeInterval 10;
        writeControl    timeStep;
        writeInterval   10;
        
        // Output options
        writeFields     true;
        writeStatistics true;
        
        // Debugging
        debug           false;
    }
    
    pressureFilter
    {
        type            myFilter;
        libs            ("libmyFunctionObjects.so");
        
        fieldName       p;
        threshold       101325;
        
        executeControl  outputTime;
        writeControl    outputTime;
    }
}

// Additional solver settings
solver
{
    // Standard solver configuration
    application     simpleFoam;
    startFrom       startTime;
    startTime       0;
    stopAt          endTime;
    endTime         1000;
    deltaT          1;
    
    // Output control
    writeControl    timeStep;
    writeInterval   100;
    purgeWrite      0;
    runTimeModifiable true;
}
```

### 6.3.2 การรวมเข้ากับ Custom Solvers

สำหรับการรวมเข้าโดยตรงกับ custom solvers:

```cpp
// In main solver loop
if (runTime.outputTime())
{
    // Create and execute functionObject manually
    dictionary filterDict;
    filterDict.add("fieldName", "U");
    filterDict.add("threshold", 10.0);
    
    autoPtr<myFunctionObject> filter = 
        myFunctionObject::New("velocityFilter", runTime, filterDict);
    
    if (filter->execute())
    {
        filter->write();
    }
}
```

## 6.4 การปรับปรุงประสิทธิภาพ

### 6.4.1 การจัดการหน่วยความจำ

```cpp
class optimizedFunctionObject
:
    public functionObject
{
private:
    // Use static memory for frequently accessed objects
    static const label minBufferSize = 1000;
    
    // Pre-allocated temporary storage
    scalarField bufferField_;
    
    // Reference to mesh (no copying)
    const fvMesh& mesh_;
    
public:
    // Move constructor for efficient transfer
    optimizedFunctionObject(optimizedFunctionObject&& other) noexcept
    :
        functionObject(std::move(other)),
        bufferField_(std::move(other.bufferField_)),
        mesh_(other.mesh_)
    {}
};
```

### 6.4.2 การดำเนินการแบบ Vectorized

```cpp
// Optimized filtering using OpenFOAM's vectorized operations
void optimizedFilter()
{
    const volScalarField& source = mesh_.lookupObject<volScalarField>(fieldName_);
    volScalarField& filtered = filteredField_();
    
    // Vectorized threshold operation
    filtered = max(source - threshold_, scalar(0)) * pos(source - threshold_);
    
    // Parallel reduction for statistics
    reducedStats_ = gSum
    (
        filtered.primitiveField() * mesh_.V()
    );
}
```
